function getNowDateStr() {
  let now = new Date();
  let nowStr =
    now.getFullYear() +
    "-" +
    (now.getMonth() + 1) +
    "-" +
    now.getDate() +
    " " +
    now.getHours() +
    ":" +
    now.getMinutes() +
    ":" +
    now.getSeconds() +
    " " +
    now.getMilliseconds();
  return nowStr;
}
